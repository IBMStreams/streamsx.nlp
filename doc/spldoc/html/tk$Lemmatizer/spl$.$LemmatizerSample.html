 					mutable list<rstring> pos = [];

 					com.ibm.streamsx.text.utils::lemmatize(text, words, pos, lemmas);

 						printStringLn ("word[" + (rstring)i + "]=" + w + " pos[" + (rstring)i + "]=" + pos[i] + " lemma[" + (rstring)i + "]=" + lemmas[i]);

 					}
